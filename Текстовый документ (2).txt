Имеем структуру проекта

TraderBook/
├── run.py                # Главный скрипт для запуска приложения
├── requirements.txt 
└── src/                  # Основной код приложения
    ├── __init__.py       
    ├── trader_journal.py # Главный модуль интерфейса
    ├── core/             # Модули ядра
    │   ├── __init__.py      
    │   ├── config.py     # Конфигурационные константы
    │   ├── logging_setup.py # Настройка логгирования
    │   ├── data_manager.py # Управление данными (trades, setups)
    │   ├── backup_manager.py # Управление резервными копиями
    │   ├── file_manager.py # Инициализация файлов
    │   └── calculations.py # Расчёты для сделок
    └── ui/               # Модули пользовательского интерфейса
        ├── __init__.py       
        ├── sidebar.py    # Боковая панель (язык, часовой пояс, импорт)
        ├── trade_input.py # Вкладка ввода сделок (Крипто/Форекс)
        ├── trade_list.py # Вкладка списка сделок
        ├── setups.py     # Вкладка управления сетапами
        ├── analytics.py  # Вкладка аналитики
        ├── translations.py # Загрузка переводов
        └── utils.py      # Утилиты (валидация, скриншоты)
    └── translations/     # Переводы
        ├── ru.json       # Русский перевод
        ├── uk.json       # Украинский перевод
        └── en.json       # Английский перевод

При запуске проекта формируется такая файловая структура в корне проекта, т.е. по пути C:\TraderBook

├── data/                 # Папка для данных
     ├── trades.csv        # Файл с данными сделок
     ├── setups.csv        # Файл с данными сетапов
     ├── language.txt        # Файл с последним выбранным языком в приложении
     ├── reward_risk.txt     # Файл с риск-ревардом по умолчанию
     ├── timezone.txt        # Файл с установленным смещением времени
     └──logs/                 # Папка для логов
        └── app.log           # Лог-файл приложения
        └── backup.log        # Лог-файл создания резервных копий
        └── CSV.log           # Лог-файл создания/загрузки списка сделок
     ├── screenshots/          # Папка для скриншотов сделок
           └──screenshots_entry/   # Папка для скриншотов входов по сделкам 
           └──screenshots_exit/   # Папка для скриншотов выходов по сделкам 
     ├── setup_images/         # Папка для изображений сетапов
     ├── backups/
          └── backup_*.zip      # Резервные копии (динамическое имя)
